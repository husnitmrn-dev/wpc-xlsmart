1. Buat Table & Bucket - Copy SQL ini ke Supabase SQL Editor:
-- Buat table
create table if not exists wpc_manpower (
  id bigint primary key,
  nama text not null,
  posisi text not null,
  status text not null,
  project text not null,
  subcont text not null,
  link_cv text,
  file_name text,
  file_url text,
  catatan text,
  tgl text,
  created_at timestamp default now()
);
-- Enable RLS
alter table wpc_manpower enable row level security;
create policy "Allow all" on wpc_manpower for all using (true) with check (true);

-- Buat storage bucket untuk CV
insert into storage.buckets (id, name, public) values ('cv-files', 'cv-files', true) on conflict (id) do nothing;
create policy "Public Access" on storage.objects for all using (bucket_id='cv-files') with check (bucket_id='cv-files');
